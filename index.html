export default { async fetch(request, env) { const url = new URL(request.url); const origin = request.headers.get('Origin') || '*'; // CORS preflight if (request.method === 'OPTIONS') { return new Response(null, { headers: { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type, Authorization, X-Discogs-Token, X-EB-User, X-EB-Token, X-Bertus-Token', } }); } const cors = { 'Access-Control-Allow-Origin': '*', 'Content-Type': 'application/json', }; const path = url.pathname; // ── Discogs ────────────────────────────────────────────── if (path.startsWith('/discogs/')) { const token = request.headers.get('X-Discogs-Token'); if (!token) return json({ error: 'Geen Discogs token' }, 401, cors); const discogsPath = path.replace('/discogs/', ''); const params = url.searchParams.toString(); const discogsUrl = `https://api.discogs.com/${discogsPath}${params ? '?' + params : ''}`; const resp = await fetch(discogsUrl, { headers: { 'Authorization': `Discogs token=${token}`, 'User-Agent': 'Major9VinylDashboard/1.0 +major9.nl', 'Accept': 'application/json', } }); const data = await resp.text(); return new Response(data, { status: resp.status, headers: { ...cors, 'Content-Type': 'application/json' } }); } // ── e-Boekhouden ───────────────────────────────────────── if (path.startsWith('/eboekhouden/')) { const ebUser = request.headers.get('X-EB-User'); const ebToken = request.headers.get('X-EB-Token'); if (!ebUser || !ebToken) return json({ error: 'Geen e-Boekhouden credentials' }, 401, cors); const subpath = path.replace('/eboekhouden/', ''); // e-Boekhouden uses SOAP — we build a minimal SOAP envelope if (subpath === 'mutaties') { const soapBody = ` ${ebUser} ${ebToken} vinyl 2024-01-01 2026-12-31 `; const resp = await fetch('https://soap.e-boekhouden.nl/boekhouden/webservice.asmx', { method: 'POST', headers: { 'Content-Type': 'text/xml; charset=utf-8', 'SOAPAction': 'https://soap.e-boekhouden.nl/boekhouden/webservice/GetMutaties', }, body: soapBody, }); const xml = await resp.text(); // Parse SOAP XML to JSON const mutaties = parseMutatiesXML(xml); return new Response(JSON.stringify({ mutaties }), { status: 200, headers: { ...cors, 'Content-Type': 'application/json' } }); } // Session-based auth attempt if (subpath === 'session') { const soapBody = ` ${ebUser} ${ebToken} `; const resp = await fetch('https://soap.e-boekhouden.nl/boekhouden/webservice.asmx', { method: 'POST', headers: { 'Content-Type': 'text/xml; charset=utf-8', 'SOAPAction': 'https://soap.e-boekhouden.nl/boekhouden/webservice/OpenSession', }, body: soapBody, }); const xml = await resp.text(); const sessionMatch = xml.match(/([^<]+)<\/SessionID>/); const sessionID = sessionMatch ? sessionMatch[1] : null; return new Response(JSON.stringify({ sessionID, raw: xml.slice(0, 500) }), { headers: { ...cors } }); } return json({ error: 'Onbekend e-Boekhouden endpoint: ' + subpath }, 404, cors); } // ── Bertus ─────────────────────────────────────────────── if (path.startsWith('/bertus/')) { const token = request.headers.get('X-Bertus-Token'); if (!token) return json({ error: 'Geen Bertus token' }, 401, cors); const subpath = path.replace('/bertus/', ''); const params = url.searchParams.toString(); const bertusUrl = `https://api.bertus.nl/v1/${subpath}${params ? '?' + params : ''}`; const resp = await fetch(bertusUrl, { headers: { 'Authorization': `Bearer ${token}`, 'Accept': 'application/json', 'User-Agent': 'Major9VinylDashboard/1.0', } }); const data = await resp.text(); return new Response(data, { status: resp.status, headers: { ...cors, 'Content-Type': 'application/json' } }); } // ── Status check ───────────────────────────────────────── if (path === '/' || path === '') { return new Response(JSON.stringify({ status: 'Major9 Vinyl Worker actief', endpoints: ['/discogs/*', '/eboekhouden/mutaties', '/bertus/*'], version: '1.0' }), { headers: cors }); } return json({ error: 'Endpoint niet gevonden: ' + path }, 404, cors); } }; function json(obj, status = 200, headers = {}) { return new Response(JSON.stringify(obj), { status, headers: { 'Content-Type': 'application/json', ...headers } }); } function parseMutatiesXML(xml) { const items = []; const matches = xml.matchAll(/]*>([\s\S]*?)<\/cMutatieList>/g); for (const match of matches) { const block = match[1]; const get = (tag) => { const m = block.match(new RegExp(`<${tag}>([^<]*)<\/${tag}>`)); return m ? m[1].trim() : ''; }; items.push({ omschrijving: get('Omschrijving') || get('InvoiceNumber') || '—', bedrag: parseFloat(get('BedragDebet') || get('BedragCredit') || '0'), datum: get('Datum') ? get('Datum').slice(0, 10) : '', categorie: get('Categorie') || get('GrootboekOmschrijving') || 'Overig', grootboek: get('GrootboekCode') || '', }); } return items; }
